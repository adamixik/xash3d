cmake_minimum_required(VERSION 2.6.0)
project (XASH_ENGINE)

# set (CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -Wall -Wextra -pedantic")
set (CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -w")
set (CMAKE_CXX_FLAGS_DEBUG "${CMAKE_CXX_FLAGS_DEBUG} -g")
set (CMAKE_CXX_FLAGS_RELEASE "${CMAKE_CXX_FLAGS_RELEASE} -O3")
set (CMAKE_CXX_FLAGS_MINSIZEREL "${CMAKE_CXX_FLAGS_MINSIZEREL} -Os")
set (CMAKE_CXX_FLAGS_RELWITHDEBINFO "${CMAKE_CXX_FLAGS_RELWITHDEBINFO} -g -O3")

set (CMAKE_C_FLAGS "${CMAKE_CXX_FLAGS}")
set (CMAKE_C_FLAGS_DEBUG "${CMAKE_CXX_FLAGS_DEBUG}")
set (CMAKE_C_FLAGS_RELEASE "${CMAKE_CXX_FLAGS_RELEASE}")
set (CMAKE_C_FLAGS_MINSIZEREL "${CMAKE_CXX_FLAGS_MINSIZEREL}")
set (CMAKE_C_FLAGS_RELWITHDEBINFO "${CMAKE_CXX_FLAGS_RELWITHDEBINFO}")

if(PANDORA)
        add_definitions(-DPANDORA)
endif()

set (XASH_ENGINE_LIBRARY xash)

file (GLOB_RECURSE XASH_ENGINE_SOURCES *.cpp *.c)
file (GLOB_RECURSE XASH_ENGINE_HEADERS *.h)
include_directories (. common/
    common/imagelib/
    common/soundlib/
    client/
    client/vgui/
    server/
    ../common
    ../pm_shared
    ${HL_SDK_DIR}/public
    )

add_library (${XASH_ENGINE_LIBRARY} SHARED ${XASH_ENGINE_HEADERS} ${XASH_ENGINE_SOURCES})

if(XASH_X11)
    add_definitions(-DXASH_X11)
    target_link_libraries(${XASH_ENGINE_LIBRARY} X11)
endif()

if(XASH_IPX)
    add_definitions(-DXASH_IPX)
endif()

if(XASH_NONSTANDART_LOAD)
    add_definitions(-DXASH_NONSTANDART_LOAD)
endif()

if(XASH_SDL)
    include(FindPkgConfig)
if(WIN32)
    set(SDL2_INCLUDE_DIRS ../deps/SDL2/include)
else()
    pkg_search_module(SDL2 REQUIRED sdl2)
    pkg_search_module(SDL2_IMAGE REQUIRED SDL2_image)
    
endif()
    add_definitions(-DXASH_SDL)

    include_directories(${SDL2_INCLUDE_DIRS} ${SDL2_IMAGE_INCLUDE_DIRS})
    target_link_libraries(${XASH_ENGINE_LIBRARY} ${SDL2_LIBRARIES})
endif()



if(XASH_VGUI)
    add_definitions(-DXASH_VGUI)

if(LINK_VGUI)
    if(NOT HL_SDK_DIR)
        message_error("Please specify Half-Life 1 SDK directory with -DHL_SDK_DIR=/path/")
    endif()

    if (${CMAKE_SYSTEM_NAME} MATCHES "Darwin")
        target_link_libraries(${XASH_ENGINE_LIBRARY} -L${HL_SDK_DIR}/linux/release/ ${HL_SDK_DIR}/linux/release/vgui.dylib)
    endif()
    if (${CMAKE_SYSTEM_NAME} MATCHES "Linux")
        add_custom_command(TARGET xash PRE_LINK COMMAND ${CMAKE_COMMAND} -E copy ${HL_SDK_DIR}/linux/vgui.so $<TARGET_SONAME_FILE_DIR:xash>)
        target_link_libraries(${XASH_ENGINE_LIBRARY} -L. -l:vgui.so)
    endif()
    if(WIN32)
        target_link_libraries(${XASH_ENGINE_LIBRARY} ${HL_SDK_DIR}/utils/vgui/lib/win32_vc6/vgui.lib)
    endif()
    INCLUDE_DIRECTORIES(${HL_SDK_DIR}/utils/vgui/include)
endif()
endif()

if(XASH_GLES)
    add_definitions(-DXASH_GLES)
endif()

if(XASH_DLL_LOADER)
    add_definitions(-DDLL_LOADER)
    target_link_libraries(${XASH_ENGINE_LIBRARY} libloader.so)
endif()
if(XASH_DEDICATED)
    add_definitions(-DXASH_DEDICATED)
endif()

target_link_libraries(${XASH_ENGINE_LIBRARY}
    ${CMAKE_DL_LIBS} SDL2-static)

set_target_properties (${XASH_ENGINE_SHARED} PROPERTIES
    VERSION ${XASH3D_VERSION} SOVERSION ${XASH3D_VERSION}
    POSITION_INDEPENDENT_CODE 1
)
